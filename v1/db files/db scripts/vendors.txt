CREATE TABLE vendors (

id INT(6) UNSIGNED AUTO_INCREMENT PRIMARY KEY,

firstname VARCHAR(30) NOT NULL,

lastname VARCHAR(30),
mobile VARCHAR(15),

email VARCHAR(20),

password VARCHAR(20),

city VARCHAR(20),

address VARCHAR(50),

lat VARCHAR(20),

longd VARCHAR(20),

pinc VARCHAR(10),

shop_name VARCHAR(50),

status VARCHAR(20),

verifed VARCHAR(10),

reg_date TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP

)
